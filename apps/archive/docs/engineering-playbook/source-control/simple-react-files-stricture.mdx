# Simple React Files Structure

```mermaid
graph LR
  subgraph srcBlock[src]
    src --> api1
    src --> apiN
    src --> assets
    src --> app
    src --> modules
    src --> shared
    src --> stories
    src --> test
    src --> config
  end

  subgraph appBlock[app]
    app --> App.tsx
    app --> pages
    app --> templates
    pages --> pageComponentN[ComponentN]
    templates --> templateComponentN[ComponentN]
  end

  subgraph apiBlock[api]
    apiN --> client.ts
    apiN --> apiEntityN
    apiEntityN --> api.ts
    apiEntityN --> apiEntityHooks[hooks]
    apiEntityN --> apiEntityParsers[parsers]
    apiEntityN --> apiEntityTypes[types]
    apiEntityN --> apiEntityCacheUpdates[cacheUpdates]
    apiEntityTypes --> payloads.ts
    apiEntityTypes --> responses.ts
    apiEntityTypes --> common.ts
  end

  subgraph modulesBlock[modules]
    modules --> moduleN
    moduleN --> entityN
    entityN --> entityComponents[components]
    entityN --> entityHooks[hooks]
    entityN --> entityHocs[hocs]
    entityN --> entityUtils[utils]
    entityN --> entityTypes[types]
    entityN --> entityConstants[constants]
    entityN --> entityContext[context]
  end

  subgraph sharedBlock[shared]
    shared --> groupN
    groupN --> similarToEntityStructure
  end

  subgraph componentsBlock[components]
    entityComponents --> atoms
    entityComponents --> molecules
    entityComponents --> organisms
    atoms --> Component1
    molecules --> Component2
    organisms --> ComponentN
  end

  subgraph componentBlock[Component]
    ComponentN --> ComponentN.tsx
    ComponentN --> componentTests[__tests__/integration]
    ComponentN --> componentHooks[hooks]
    ComponentN --> componentTypes[types]
    ComponentN --> componentConstants[constants]
    ComponentN --> componentContext[context]
    ComponentN --> componentComponents[components]
    ComponentN --> componentStories[stories]
    componentStories --> ComponentN.stories.tsx
    ComponentN --> componentUtils[utils]
  end

  subgraph hookBlock[hook]
    componentHooks --> useFunctionalityN
    useFunctionalityN --> useFunctionalityN.ts
    useFunctionalityN --> hookUtils[utils]
    useFunctionalityN --> hookTypes[types]
    useFunctionalityN --> hookConstants[constants]
    useFunctionalityN --> hookTests[__tests__/unit]
  end

  subgraph utilBlock[util]
    hookUtils --> utilN
    utilN --> utilN.ts
    utilN --> utilTypes[types]
    utilN --> utilConstants[constants]
    utilN --> utilTests[__tests__/unit]
  end

  subgraph contextBlock[Context]
    entityContext --> ContextN
    ContextN --> similarToComponentStructure
  end

  style srcBlock fill:lightgreen
  style apiBlock fill:skyblue
  style appBlock fill:#b5e4ff
  style modulesBlock fill:lightyellow
  style sharedBlock fill:#cfb5ff
  style componentsBlock fill:lightblue
  style componentBlock fill:#b5fff1
  style hookBlock fill:#ffdab5
  style utilBlock fill:pink
  style contextBlock fill:#cfffb5
```
